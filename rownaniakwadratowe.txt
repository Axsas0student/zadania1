#include <stdio.h>
#include <math.h>

#zmiana w nowy branch

#pragma warning (disable: 4996)

int main() {
    double a, b, c, x1, x2;

    printf("Podaj wspolczynniki a, b, c: ");
    scanf("%lf %lf %lf", &a, &b, &c);

    if (a != 0) {
        double delta = b * b - 4 * a * c;

        if (delta > 0) {
            x1 = (-b - sqrt(delta)) / (2 * a);
            x2 = (-b + sqrt(delta)) / (2 * a);
            printf("4, x1 = %lf, x2 = %lf\n", x1, x2);
        }
        else if (delta == 0) {
            x1 = -b / (2 * a);
            printf("5, x = %lf\n", x1);
        }
        else {
            printf("6\n");
        }
    }
    else {
        if (b != 0) {
            x1 = -c / b;
            printf("3, x = %lf\n", x1);
        }
        else {
            if (c == 0) {
                printf("1\n");
            }
            else {
                printf("2\n");
            }
        }
    }

    return 0;
}
